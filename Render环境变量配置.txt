# Render 环境变量配置
# 复制以下内容到 Render Dashboard 的 Environment Variables

# ============================================
# 基础配置
# ============================================
NODE_ENV=production
PORT=3000
TZ=Asia/Shanghai

# ============================================
# 数据库配置（Supabase Connection Pooler）
# ============================================
DB_HOST=aws-0-ap-southeast-1.pooler.supabase.com
DB_PORT=6543
DB_NAME=postgres
DB_USER=postgres.你的项目ID
DB_PASSWORD=你的密码

# 如何获取：
# 1. 访问 Supabase Dashboard
# 2. Settings → Database → Connection Pooling
# 3. 模式选择 Transaction
# 4. 项目ID在 Settings → General → Reference ID

# ============================================
# JWT 配置
# ============================================
JWT_SECRET=<运行命令生成: openssl rand -base64 32>
JWT_EXPIRES_IN=7d

# 生成 JWT Secret:
# 在终端运行: openssl rand -base64 32
# 复制输出结果填入上方

# ============================================
# LLM 配置（二选一）
# ============================================

# 选项 1: 使用 OpenAI
OPENAI_API_KEY=你的OpenAI密钥
OPENAI_MODEL=gpt-4

# 选项 2: 使用自定义 LLM API（如 Webull 内部 API）
LLM_API_URL=https://office.webullbroker.com/api/oa-ai/open/chat/completions
LLM_API_KEY=dcb0d1c8-82c7-478f-bd9a-7c3023cac4d8
LLM_MODEL=us.anthropic.claude-sonnet-4-20250514-v1:0
LLM_API_KEY_HEADER=authorization
LLM_AUTH_PREFIX=Bearer

# 注意：如果同时配置了 OpenAI 和自定义 LLM，
# 系统会优先使用自定义 LLM 配置

# ============================================
# Jina Reader 配置
# ============================================
JINA_READER_API_URL=https://r.jina.ai

# ============================================
# 其他配置
# ============================================
MAX_PARALLEL_WEBSITES=5
RETRIEVAL_TIMEOUT_MS=30000
ENABLE_WEBSITE_ANALYZER=true
DEMO_MODE=false

# ============================================
# 配置说明
# ============================================

# MAX_PARALLEL_WEBSITES: 同时检索的最大网站数
# RETRIEVAL_TIMEOUT_MS: 单个网站检索超时时间（毫秒）
# ENABLE_WEBSITE_ANALYZER: 是否启用网站分析器
# DEMO_MODE: 演示模式（设为 true 会使用模拟数据）

# ============================================
# 重要提示
# ============================================

# 1. 不需要 REDIS_HOST、REDIS_PORT、REDIS_PASSWORD
#    因为使用 pg-boss（基于 PostgreSQL）作为任务队列

# 2. Start Command 必须是: node dist/index.pgboss.js

# 3. 确保数据库迁移已在本地运行成功

# 4. 所有环境变量都必须填写，否则服务可能无法启动
